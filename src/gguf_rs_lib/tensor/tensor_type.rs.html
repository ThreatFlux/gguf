19/// Extended tensor type information with additional metadata
20#[cfg_attr(feature = "std", derive(Serialize, Deserialize))]
21#[derive(Debug, Clone, PartialEq)]
22pub struct TensorTypeInfo {
23    /// The tensor type
24    pub tensor_type: TensorType,
25    /// Human-readable name
26    pub name: String,
27    /// Whether this type is quantized
28    pub is_quantized: bool,
29    /// Block size for quantized types (1 for non-quantized)
30    pub block_size: usize,
31    /// Bits per weight (approximate for quantized types)
32    pub bits_per_weight: f32,
33    /// Category of quantization
34    pub quantization_category: QuantizationCategory,
35}
36
37/// Categories of quantization schemes
38#[cfg_attr(feature = "std", derive(Serialize, Deserialize))]
39#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
40pub enum QuantizationCategory {
41    /// No quantization (F32, F16, BF16, etc.)
42    None,
43    /// Legacy GGML quantization (Q4_0, Q4_1, Q5_0, Q5_1, Q8_0, Q8_1)
44    Legacy,
45    /// K-quant schemes (Q2_K through Q8_K)
46    KQuant,
47    /// IQ-quant schemes (ultra-low bit)
48    IQuant,
49    /// Integer types
50    Integer,
51}
52
53impl TensorTypeInfo {
54    /// Get tensor type information for a given type
55    pub fn for_type(tensor_type: TensorType) -> Self {
56        let name = tensor_type.name().to_string();
57        let is_quantized = tensor_type.is_quantized();
58        let block_size = tensor_type.block_size();
59
60        let (bits_per_weight, quantization_category) = match tensor_type {
61            // Floating point types
62            TensorType::F32 => (32.0, QuantizationCategory::None),
63            TensorType::F64 => (64.0, QuantizationCategory::None),
64            TensorType::F16 => (16.0, QuantizationCategory::None),
65            TensorType::BF16 => (16.0, QuantizationCategory::None),
66
67            // Integer types
68            TensorType::I8 => (8.0, QuantizationCategory::Integer),
69            TensorType::I16 => (16.0, QuantizationCategory::Integer),
70            TensorType::I32 => (32.0, QuantizationCategory::Integer),
71            TensorType::I64 => (64.0, QuantizationCategory::Integer),
72
73            // Legacy quantization
74            TensorType::Q4_0 | TensorType::Q4_1 => (4.0, QuantizationCategory::Legacy),
75            TensorType::Q4_2 | TensorType::Q4_3 => (4.0, QuantizationCategory::Legacy),
76            TensorType::Q5_0 | TensorType::Q5_1 => (5.0, QuantizationCategory::Legacy),
77            TensorType::Q8_0 | TensorType::Q8_1 => (8.0, QuantizationCategory::Legacy),
78
79            // K-quant
80            TensorType::Q2_K => (2.0, QuantizationCategory::KQuant),
81            TensorType::Q3_K => (3.0, QuantizationCategory::KQuant),
82            TensorType::Q4_K => (4.0, QuantizationCategory::KQuant),
83            TensorType::Q5_K => (5.0, QuantizationCategory::KQuant),
84            TensorType::Q6_K => (6.0, QuantizationCategory::KQuant),
85            TensorType::Q8_K => (8.0, QuantizationCategory::KQuant),
86
87            // IQ-quant (ultra-low bit)
88            TensorType::IQ1_S | TensorType::IQ1_M => (1.0, QuantizationCategory::IQuant),
89            TensorType::IQ2_XXS | TensorType::IQ2_XS | TensorType::IQ2_S => {
90                (2.0, QuantizationCategory::IQuant)
91            }
92            TensorType::IQ3_XXS | TensorType::IQ3_S => (3.0, QuantizationCategory::IQuant),
93            TensorType::IQ4_NL | TensorType::IQ4_XS | TensorType::IQ4_UNI => {
94                (4.0, QuantizationCategory::IQuant)
95            }
96        };
97
98        Self {
99            tensor_type,
100            name,
101            is_quantized,
102            block_size,
103            bits_per_weight,
104            quantization_category,
105        }
106    }
107
108    /// Check if this tensor type supports specific operations
109    pub fn supports_fast_inference(&self) -> bool {
110        // Non-quantized types generally support fast inference
111        matches!(
112            self.quantization_category,
113            QuantizationCategory::None | QuantizationCategory::Integer
114        )
115    }
116
117    /// Check if this tensor type is optimized for memory usage
118    pub fn is_memory_optimized(&self) -> bool {
119        self.bits_per_weight < 16.0
120    }
121
122    /// Get the compression ratio compared to F32
123    pub fn compression_ratio(&self) -> f32 {
124        32.0 / self.bits_per_weight
125    }
126
127    /// Get the theoretical memory savings compared to F32
128    pub fn memory_savings(&self) -> f32 {
129        1.0 - (self.bits_per_weight / 32.0)
130    }
131
132    /// Check if this type is suitable for a given precision requirement
133    pub fn meets_precision_requirement(&self, min_bits: f32) -> bool {
134        self.bits_per_weight >= min_bits
135    }
136}

141        write!(
142            f,
143            "{} ({:.1} bits/weight, {}x compression, {:?})",

149    }

163    }

208    /// Get all quantized tensor types
209    pub fn quantized_types() -> Vec<TensorType> {
210        Self::all_types().into_iter().filter(|t| t.is_quantized()).collect()
211    }

213    /// Get all non-quantized tensor types
214    pub fn non_quantized_types() -> Vec<TensorType> {
215        Self::all_types().into_iter().filter(|t| !t.is_quantized()).collect()
216    }

218    /// Get tensor types by category
219    pub fn types_by_category(category: QuantizationCategory) -> Vec<TensorType> {
220        Self::all_types()
221            .into_iter()
222            .filter(|t| {
223                let info = TensorTypeInfo::for_type(*t);
224                info.quantization_category == category
225            })
226            .collect()
227    }

229    /// Find the best tensor type for given requirements
230    pub fn find_best_type(
231        max_bits: f32,
232        prefer_k_quant: bool,
233        require_fast_inference: bool,
234    ) -> Option<TensorType> {
235        let mut candidates: Vec<_> = Self::all_types()
236            .into_iter()
237            .map(|t| (t, TensorTypeInfo::for_type(t)))
238            .filter(|(_, info)| info.bits_per_weight <= max_bits)
239            .collect();
240
241        if require_fast_inference {
242            candidates.retain(|(_, info)| info.supports_fast_inference());
243        }
244
245        if candidates.is_empty() {
246            return None;
247        }
248
249        // Sort by preference
250        candidates.sort_by(|(_, a), (_, b)| {
251            // Prefer K-quant if requested
252            if prefer_k_quant {
253                match (a.quantization_category, b.quantization_category) {
254                    (QuantizationCategory::KQuant, QuantizationCategory::KQuant) => {}
255                    #[cfg(feature = "std")]
256                    (QuantizationCategory::KQuant, _) => return std::cmp::Ordering::Less,
257                    #[cfg(feature = "std")]
258                    (_, QuantizationCategory::KQuant) => return std::cmp::Ordering::Greater,
259                    #[cfg(not(feature = "std"))]
260                    (QuantizationCategory::KQuant, _) => return Ordering::Less,
261                    #[cfg(not(feature = "std"))]
262                    (_, QuantizationCategory::KQuant) => return Ordering::Greater,
263                    _ => {}
264                }
265            }
266
267            // Then prefer higher precision within constraints
268            b.bits_per_weight.partial_cmp(&a.bits_per_weight).unwrap()
269        });
270
271        candidates.first().map(|(t, _)| *t)
272    }
273
274    /// Check if a tensor type is deprecated
275    pub fn is_deprecated(tensor_type: TensorType) -> bool {
276        matches!(tensor_type, TensorType::Q4_2 | TensorType::Q4_3)
277    }
278
279    /// Get the recommended replacement for a deprecated type
280    pub fn get_replacement(tensor_type: TensorType) -> Option<TensorType> {
281        match tensor_type {
282            TensorType::Q4_2 | TensorType::Q4_3 => Some(TensorType::Q4_0),
283            _ => None,
284        }
285    }
286
287    /// Get tensor types suitable for a model size
288    pub fn types_for_model_size(model_size_gb: f32) -> Vec<TensorType> {
289        let mut types = Vec::new();
290
291        // For small models, all types are fine
292        if model_size_gb < 1.0 {
293            return Self::all_types();
294        }
295
296        // For medium models, prefer more aggressive quantization
297        if model_size_gb < 7.0 {
298            types.extend_from_slice(&[
299                TensorType::Q4_0,
300                TensorType::Q4_1,
301                TensorType::Q5_0,
302                TensorType::Q5_1,
303                TensorType::Q4_K,
304                TensorType::Q5_K,
305                TensorType::Q6_K,
306                TensorType::F16,
307                TensorType::BF16,
308            ]);
309        }
310
311        // For large models, prefer very aggressive quantization
312        if model_size_gb < 30.0 {
313            types.extend_from_slice(&[
314                TensorType::Q2_K,
315                TensorType::Q3_K,
316                TensorType::Q4_K,
317                TensorType::IQ2_XS,
318                TensorType::IQ3_S,
319                TensorType::IQ4_NL,
320            ]);
321        } else {
322            // For very large models, use the most aggressive quantization
323            types.extend_from_slice(&[
324                TensorType::IQ1_S,
325                TensorType::IQ1_M,
326                TensorType::IQ2_XXS,
327                TensorType::IQ2_XS,
328                TensorType::IQ3_XXS,
329            ]);
330        }
331
332        types
333    }

398    fn test_find_best_type() {
399        // Test with high precision requirement
400        let best = TensorTypeUtils::find_best_type(8.0, false, false);
401        assert!(best.is_some());
402        let info = TensorTypeInfo::for_type(best.unwrap());
403        assert!(info.bits_per_weight <= 8.0);
404
405        // Test with K-quant preference
406        let best_k = TensorTypeUtils::find_best_type(8.0, true, false);
407        if let Some(best_type) = best_k {
408            let info = TensorTypeInfo::for_type(best_type);
409            if info.bits_per_weight <= 8.0 {
410                // Should prefer K-quant if available in range
411                let k_quants = TensorTypeUtils::types_by_category(QuantizationCategory::KQuant);
412                let has_k_in_range = k_quants.iter().any(|t| {
413                    let t_info = TensorTypeInfo::for_type(*t);
414                    t_info.bits_per_weight <= 8.0
415                });
416                if has_k_in_range {
417                    assert_eq!(info.quantization_category, QuantizationCategory::KQuant);
418                }
419            }
420        }
