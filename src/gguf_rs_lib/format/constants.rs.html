1//! GGUF format constants and magic numbers
2
3/// Magic number for GGUF files ("GGUF" in little-endian byte order)
4pub const GGUF_MAGIC: u32 = 0x4655_4747;
5
6/// Current GGUF format version
7pub const GGUF_VERSION: u32 = 3;
8
9/// Default alignment requirement for tensor data (32 bytes)
10pub const GGUF_DEFAULT_ALIGNMENT: usize = 32;
11
12/// Maximum supported string length in metadata
13pub const GGUF_MAX_STRING_LENGTH: usize = 65_536;
14
15/// Maximum supported array length in metadata
16pub const GGUF_MAX_ARRAY_LENGTH: usize = 65_536;
17
18/// Size of the GGUF header in bytes (magic + version + tensor_count + metadata_kv_count)
19pub const GGUF_HEADER_SIZE: usize = 4 + 4 + 8 + 8;
20
21/// Minimum valid GGUF file size (header only)
22pub const GGUF_MIN_FILE_SIZE: usize = GGUF_HEADER_SIZE;

28    #[test]

30        // Verify the magic number is "GGUF" in little-endian

32        assert_eq!(magic_bytes, [b'G', b'G', b'U', b'F']);
33    }
34
35    #[test]
36    fn test_constants() {

38        assert_eq!(GGUF_DEFAULT_ALIGNMENT, 32);
39        assert_eq!(GGUF_HEADER_SIZE, 24);
40    }
