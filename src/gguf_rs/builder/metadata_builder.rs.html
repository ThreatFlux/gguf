1//! Metadata builder utilities
2
3use crate::format::{metadata::MetadataValue, Metadata};
4
5#[cfg(not(feature = "std"))]
6extern crate alloc;
7#[cfg(not(feature = "std"))]
8use alloc::string::String;
9
10/// Builder for GGUF metadata
11#[derive(Debug, Default, Clone)]
12pub struct MetadataBuilder {
13    metadata: Metadata,
14}
15
16impl MetadataBuilder {
17    /// Create a new metadata builder
18    pub fn new() -> Self {
19        Self::default()
20    }
21
22    /// Add a string value
23    pub fn add_string<K: Into<String>, V: Into<String>>(mut self, key: K, value: V) -> Self {
24        self.metadata.insert(key.into(), MetadataValue::String(value.into()));
25        self
26    }
27
28    /// Add a u32 value
29    pub fn add_u32<K: Into<String>>(mut self, key: K, value: u32) -> Self {
30        self.metadata.insert(key.into(), MetadataValue::U32(value));
31        self
32    }
33
34    /// Add a u64 value
35    pub fn add_u64<K: Into<String>>(mut self, key: K, value: u64) -> Self {
36        self.metadata.insert(key.into(), MetadataValue::U64(value));
37        self
38    }
39
40    /// Add a f32 value
41    pub fn add_f32<K: Into<String>>(mut self, key: K, value: f32) -> Self {
42        self.metadata.insert(key.into(), MetadataValue::F32(value));
43        self
44    }
45
46    /// Add a boolean value
47    pub fn add_bool<K: Into<String>>(mut self, key: K, value: bool) -> Self {
48        self.metadata.insert(key.into(), MetadataValue::Bool(value));
49        self
50    }
51
52    /// Add any metadata value
53    pub fn add<K: Into<String>>(mut self, key: K, value: MetadataValue) -> Self {
54        self.metadata.insert(key.into(), value);
55        self
56    }
57
58    /// Build the metadata
59    pub fn build(self) -> Metadata {
60        self.metadata
61    }
62
63    /// Get the current size
64    pub fn len(&self) -> usize {
65        self.metadata.len()
66    }
67
68    /// Check if empty
69    pub fn is_empty(&self) -> bool {
70        self.metadata.is_empty()
71    }
72}
73
74/// Common metadata builders for standard model types
75impl MetadataBuilder {
76    /// Create metadata for a language model
77    pub fn language_model<N: Into<String>>(name: N) -> Self {
78        Self::new()
79            .add_string("general.architecture", "llama")
80            .add_string("general.name", name)
81            .add_u32("general.file_type", 1)
82    }
83
84    /// Create metadata for a vision model
85    pub fn vision_model<N: Into<String>>(name: N) -> Self {
86        Self::new()
87            .add_string("general.architecture", "clip")
88            .add_string("general.name", name)
89            .add_u32("general.file_type", 1)
90    }
91
92    /// Add common LLaMA parameters
93    pub fn with_llama_params(
94        self,
95        context_length: u64,
96        embedding_length: u64,
97        vocab_size: u64,
98    ) -> Self {
99        self.add_u64("llama.context_length", context_length)
100            .add_u64("llama.embedding_length", embedding_length)
101            .add_u64("llama.vocab_size", vocab_size)
102    }
103}

105#[cfg(all(test, feature = "std"))]
106mod tests {
107    use super::*;
108
109    #[test]
110    fn test_metadata_builder() {
111        let metadata = MetadataBuilder::new()
112            .add_string("name", "test")
113            .add_u32("version", 1)
114            .add_bool("enabled", true)
115            .build();
116
117        assert_eq!(metadata.len(), 3);
118        assert_eq!(metadata.get_string("name"), Some("test"));
119        assert_eq!(metadata.get_u64("version"), Some(1));
120        assert_eq!(metadata.get_bool("enabled"), Some(true));
121    }
122
123    #[test]
124    fn test_language_model_metadata() {
125        let metadata = MetadataBuilder::language_model("test_llama")
126            .with_llama_params(2048, 4096, 32000)
127            .build();
128
129        assert_eq!(metadata.get_string("general.architecture"), Some("llama"));
130        assert_eq!(metadata.get_u64("llama.context_length"), Some(2048));
131    }
