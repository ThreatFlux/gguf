25/// GGUF file header structure
26#[derive(Debug, Clone, PartialEq, Eq)]
27pub struct GGUFHeader {
28    /// Magic number ("GGUF" in little-endian)
29    pub magic: u32,
30    /// Version number (currently 3)
31    pub version: u32,
32    /// Number of tensors in the file
33    pub tensor_count: u64,
34    /// Number of metadata key-value pairs
35    pub metadata_kv_count: u64,
36}
37
38impl Default for GGUFHeader {
39    fn default() -> Self {
40        Self::new(0, 0)
41    }
42}
43
44impl GGUFHeader {
45    /// Create a new GGUF header with default values
46    pub fn new(tensor_count: u64, metadata_kv_count: u64) -> Self {
47        Self { magic: GGUF_MAGIC, version: GGUF_VERSION, tensor_count, metadata_kv_count }
48    }
49
50    /// Check if the header has valid magic number and version
51    pub fn is_valid(&self) -> bool {
52        self.magic == GGUF_MAGIC && self.version == GGUF_VERSION
53    }
54
55    /// Validate the header and return an error if invalid
56    pub fn validate(&self) -> Result<()> {
57        if self.magic != GGUF_MAGIC {
58            return Err(GGUFError::InvalidMagic { expected: GGUF_MAGIC, found: self.magic });
59        }
60
61        if self.version != GGUF_VERSION {
62            return Err(GGUFError::UnsupportedVersion(self.version));
63        }
64
65        Ok(())
66    }
67
68    /// Get the size of the header in bytes
69    pub const fn size() -> usize {
70        GGUF_HEADER_SIZE
71    }
72
73    /// Read a header from a reader
74    #[cfg(feature = "std")]
75    pub fn read_from<R: Read>(reader: &mut R) -> Result<Self> {
76        let magic = read_u32(reader)?;
77        let version = read_u32(reader)?;
78        let tensor_count = read_u64(reader)?;
79        let metadata_kv_count = read_u64(reader)?;
80
81        let header = Self { magic, version, tensor_count, metadata_kv_count };
82
83        header.validate()?;
84        Ok(header)
85    }
86
87    /// Write the header to a writer
88    #[cfg(feature = "std")]
89    pub fn write_to<W: Write>(&self, writer: &mut W) -> Result<()> {
90        self.validate()?;
91
92        write_u32(writer, self.magic)?;
93        write_u32(writer, self.version)?;
94        write_u64(writer, self.tensor_count)?;
95        write_u64(writer, self.metadata_kv_count)?;
96
97        Ok(())
98    }
99
100    /// Get the total size of header + metadata + tensor info sections
101    pub fn calculate_metadata_size(&self, metadata_size: usize, tensor_info_size: usize) -> usize {
102        Self::size() + metadata_size + tensor_info_size
103    }
104
105    /// Check if the tensor count is reasonable (not too large)
106    pub fn is_tensor_count_reasonable(&self) -> bool {
107        // Reasonable limit: 1 million tensors
108        self.tensor_count <= 1_000_000
109    }
110
111    /// Check if the metadata count is reasonable (not too large)
112    pub fn is_metadata_count_reasonable(&self) -> bool {
113        // Reasonable limit: 100,000 metadata items
114        self.metadata_kv_count <= 100_000
115    }
116
117    /// Perform comprehensive validation including reasonableness checks
118    pub fn validate_comprehensive(&self) -> Result<()> {
119        self.validate()?;
120
121        if !self.is_tensor_count_reasonable() {
122            return Err(GGUFError::Format(format!(
123                "Unreasonable tensor count: {}",
124                self.tensor_count
125            )));
126        }
127
128        if !self.is_metadata_count_reasonable() {
129            return Err(GGUFError::Format(format!(
130                "Unreasonable metadata count: {}",
131                self.metadata_kv_count
132            )));
133        }
134
135        Ok(())
136    }
137}

150/// Information about a tensor in the GGUF file
151#[derive(Debug, Clone, PartialEq, Eq)]
152pub struct TensorInfo {
153    /// Name of the tensor
154    pub name: String,
155    /// Number of dimensions
156    pub n_dimensions: u32,
157    /// Shape of the tensor (dimensions)
158    pub dimensions: Vec<u64>,
159    /// Type of the tensor data
160    pub tensor_type: u32,
161    /// Offset of the tensor data from the start of the data section
162    pub offset: u64,
163}
164
165impl TensorInfo {
166    /// Create a new tensor info
167    pub fn new(name: String, dimensions: Vec<u64>, tensor_type: u32, offset: u64) -> Self {
168        let n_dimensions = dimensions.len() as u32;
169        Self { name, n_dimensions, dimensions, tensor_type, offset }
170    }
171
172    /// Calculate the number of elements in the tensor
173    pub fn element_count(&self) -> u64 {
174        self.dimensions.iter().product()
175    }
176
177    /// Get the shape as a slice
178    pub fn shape(&self) -> &[u64] {
179        &self.dimensions
180    }
181
182    /// Check if the tensor info is valid
183    pub fn is_valid(&self) -> bool {
184        // Basic validation
185        !self.name.is_empty()
186            && self.n_dimensions as usize == self.dimensions.len()
187            && self.n_dimensions > 0
188            && !self.dimensions.is_empty()
189            && self.dimensions.iter().all(|&d| d > 0)
190    }
191
192    /// Validate the tensor info
193    pub fn validate(&self) -> Result<()> {
194        if self.name.is_empty() {
195            return Err(GGUFError::InvalidTensorData("Tensor name cannot be empty".to_string()));
196        }
197
198        if self.n_dimensions as usize != self.dimensions.len() {
199            return Err(GGUFError::InvalidTensorData(format!(
200                "Dimension count mismatch: expected {}, got {}",
201                self.n_dimensions,
202                self.dimensions.len()
203            )));
204        }
205
206        if self.n_dimensions == 0 {
207            return Err(GGUFError::InvalidTensorData(
208                "Tensor must have at least one dimension".to_string(),
209            ));
210        }
211
212        // Allow zero dimensions for empty tensors - they represent tensors with 0 elements
213        // This is mathematically valid and commonly used in practice
214
215        // Check for reasonable dimension sizes (prevent integer overflow)
216        let max_dim_size = u64::MAX / (self.n_dimensions as u64 * 8); // Conservative limit
217        if self.dimensions.iter().any(|&d| d > max_dim_size) {
218            return Err(GGUFError::InvalidTensorData("Dimension size too large".to_string()));
219        }
220
221        Ok(())
222    }
223
224    /// Calculate the minimum size needed to store this tensor info
225    pub fn serialized_size(&self) -> usize {
226        // String length (8 bytes) + string data + n_dimensions (4 bytes) + dimensions + type (4 bytes) + offset (8 bytes)
227        8 + self.name.len() + 4 + (self.dimensions.len() * 8) + 4 + 8
228    }
229
230    /// Read tensor info from a reader
231    #[cfg(feature = "std")]
232    pub fn read_from<R: Read>(reader: &mut R) -> Result<Self> {
233        // Read string length and name
234        let name_len = read_u64(reader)? as usize;
235        if name_len > GGUF_MAX_STRING_LENGTH {
236            return Err(GGUFError::Format(format!("Tensor name too long: {} bytes", name_len)));
237        }
238
239        let mut name_bytes = vec![0u8; name_len];
240        reader.read_exact(&mut name_bytes)?;
241        let name = String::from_utf8(name_bytes)
242            .map_err(|e| GGUFError::Format(format!("Invalid UTF-8 in tensor name: {}", e)))?;
243
244        // Read dimensions
245        let n_dimensions = read_u32(reader)?;
246        if n_dimensions == 0 || n_dimensions > 8 {
247            return Err(GGUFError::InvalidTensorData(format!(
248                "Invalid dimension count: {}",
249                n_dimensions
250            )));
251        }
252
253        let mut dimensions = Vec::with_capacity(n_dimensions as usize);
254        for _ in 0..n_dimensions {
255            dimensions.push(read_u64(reader)?);
256        }
257
258        // Read tensor type and offset
259        let tensor_type = read_u32(reader)?;
260        let offset = read_u64(reader)?;
261
262        let tensor_info = Self { name, n_dimensions, dimensions, tensor_type, offset };
263
264        tensor_info.validate()?;
265        Ok(tensor_info)
266    }
267
268    /// Write tensor info to a writer
269    #[cfg(feature = "std")]
270    pub fn write_to<W: Write>(&self, writer: &mut W) -> Result<()> {
271        self.validate()?;
272
273        // Write string length and name
274        write_u64(writer, self.name.len() as u64)?;
275        writer.write_all(self.name.as_bytes())?;
276
277        // Write dimensions
278        write_u32(writer, self.n_dimensions)?;
279        for &dimension in &self.dimensions {
280            write_u64(writer, dimension)?;
281        }
282
283        // Write tensor type and offset
284        write_u32(writer, self.tensor_type)?;
285        write_u64(writer, self.offset)?;
286
287        Ok(())
288    }
289}
